Array in Bash Scripting
=======================

![image](https://user-images.githubusercontent.com/53966749/197399415-b3962a60-07ed-4ec9-a467-e4b56d1354b5.png)

```
[root@veena ~]# y=5
[root@veena ~]# echo $y
5
[root@veena ~]# echo ${y}
5
[root@veena ~]# x=(3 5 6.7 "ls -lrth")
[root@veena ~]# echo $x
3
[root@veena ~]# echo ${x}
3
[root@veena ~]# echo ${x[*]}
3 5 6.7 ls -lrth
[root@veena ~]# echo ${x[@]}
3 5 6.7 ls -lrth
[root@veena ~]# echo ${x[2]}
6.7
[root@veena ~]# echo ${x[-1]}
ls -lrth

```
