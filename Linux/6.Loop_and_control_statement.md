C-Syle for loop
---------------
```

[root@veena rama]# cat 4.sh
for (( i=0 ;i<=10; i++))
do
echo $i
done
```

Infinite for loop
----------------
```
[root@veena rama]# cat 4.sh
for (( ;;))
do
echo "infinite loop"
done



[root@veena rama]# cat 4.sh
cnt=1
for (( ;;))
do
echo "infinite loop $cnt"
((cnt++))
if [[  $cnt -eq 10 ]]
then
break
fi
done

[root@veena rama]# bash 4.sh
infinite loop 1
infinite loop 2
infinite loop 3
infinite loop 4
infinite loop 5
infinite loop 6
infinite loop 7
infinite loop 8
infinite loop 9

```
