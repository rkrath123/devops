root@ubuntu-3:~# cat debug.yml
---
 - hosts: localhost
   vars:
    x: 23
   tasks:
   - debug: var=x

List and Dictionary
===================

![image](https://user-images.githubusercontent.com/53966749/199137595-475878ef-4843-4b5c-ab59-ef4906cf3cee.png)

```
---
 - hosts: localhost
   vars:
    x: 34
    #pakgs: ['vim','nano','httpd','nginx']
    pakgs:
    - 'vim'
    - 'nano'
    - 'httpd'
    - 'nginx'
    #web_servers: {'Linux': 'httpd', 'ubuntu': 'apache2'}
    web_servers:
     'Linux': 'httpd'
     'ubuntu': 'apache2'
   gather_facts: false
   tasks:
   - debug: var=x
   - debug: var=pakgs
   - debug: var=pakgs[1]
   - debug: var=web_servers
   - debug: var=web_servers.keys()
   - debug: var=web_servers['Linux']
   - debug: var=web_servers.get('Linux')

root@ubuntu-3:~# ansible-playbook debug.yml

PLAY [localhost] ***************************************************************************************************************************************************************************************

TASK [debug] *******************************************************************************************************************************************************************************************
ok: [localhost] => {
    "x": 34
}

TASK [debug] *******************************************************************************************************************************************************************************************
ok: [localhost] => {
    "pakgs": [
        "vim",
        "nano",
        "httpd",
        "nginx"
    ]
}

TASK [debug] *******************************************************************************************************************************************************************************************
ok: [localhost] => {
    "pakgs[1]": "nano"
}

TASK [debug] *******************************************************************************************************************************************************************************************
ok: [localhost] => {
    "web_servers": {
        "Linux": "httpd",
        "ubuntu": "apache2"
    }
}

TASK [debug] *******************************************************************************************************************************************************************************************
ok: [localhost] => {
    "web_servers.keys()": "dict_keys(['Linux', 'ubuntu'])"
}

TASK [debug] *******************************************************************************************************************************************************************************************

    "web_servers['Linux']": "httpd"
}

TASK [debug] *******************************************************************************************************************************************************************************************
ok: [localhost] => {
    "web_servers.get('Linux')": "httpd"
}

PLAY RECAP *********************************************************************************************************************************************************************************************
localhost                  : ok=7    changed=0    unreachable=0    failed=0    skipped=0    rescued=0    ignored=0

```
